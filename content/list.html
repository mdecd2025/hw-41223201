from browser import html, document

# 讀取學員資料的 URL
url = "https://mde.tw/list/2b.txt"

# 讀取資料並將每行資料存儲到數列中
data = open(url).readlines()

# 因為資料的第一行是標題，所以有效資料從索引 1 開始
all_stud = data[1:]

# 查詢有多少學員資料
print("總共有 " + str(len(all_stud)) + "名學員")

# 取得網頁中 ID 為 "brython_div1" 的位置，這樣可以將資料插入到該位置
brython_div1 = document["brython_div1"]

# 使用 for 迴圈遍歷所有學員資料，將每個學員的資料顯示為超鏈接
for idx, student in enumerate(all_stud, start=1):
    # 假設每行資料是學員的某個識別信息，例如學號或帳號，並且以學號為鏈接名稱
    student_info = student.strip()  # 去除多餘的空格與換行符

    # 顯示學員的資料作為超鏈接
    brython_div1 <= f"{idx}. " + html.A(student_info, href=f"https://example.com/{student_info}")

    # 在每個超鏈接後插入換行符
    brython_div1 <= html.BR()
